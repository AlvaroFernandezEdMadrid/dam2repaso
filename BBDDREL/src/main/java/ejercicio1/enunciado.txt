1-> Crear tablas
2-> Crear App que haga:
	- Insertar 2 grupos.
	- Insertar 3 alumnos (No se puede meter alumnos en grupos que no existen)
	- Cambiar un alumno de grupo.
	- Borrar un alumno.
	- Borrar grupo y relocalizar alumnos.
3-> Mostrar datos de un grupo (con sus alumnos)

CREATE TABLE grupos(
nombre VARCHAR(10) PRIMARY KEY,
tutor VARCHAR(20),
curso int
);

CREATE TABLE alumnos(
nia VARCHAR(9) PRIMARY KEY,
nombre VARCHAR(30),
fecha_nacimiento DATE,
beca BOOLEAN,
grupos_nombre VARCHAR(20),
CONSTRAINT alum_grup_fk FOREIGN KEY (grupos_nombre) REFERENCES grupos(nombre)
);
